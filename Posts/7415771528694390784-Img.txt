import numpy as np
import pandas as pd
import statsmodels.api as sm
import matplotlib.pyplot as plt
import seaborn as sns
sns.set()

df = pd.read_csv('data.csv')
df.rename(columns= lambda x:x.lower(), inplace=True)

# creating dummy variables for the attendance col
df.attendance = df.attendance.map({'Yes':1, 'No':0})

# defining dependent & independent variables
y = df.gpa
x_1 = df[['sat', 'attendance']]

# adding constant of 1 (x-0)
x = sm.add_constant(x_1)

# fitting an ordinary least squares linear regression model
results = sm.OLS(y, x).fit()

# printing results
results.summary()
